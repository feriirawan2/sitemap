const axios = require('axios');

js
// generate-sitemap.js
const fs = require('fs');
const axios = require('axios');
const xmlEscape = str =>
  str.replace(/&/g, '&amp;').replace(/</g, '&lt;').replace(/>/g, '&gt;');

(async () => {
  const feedUrl = 'https://belajar-dari-hp.blogspot.com/feeds/posts/default?alt=rss';

  try {
    const res = await axios.get(feedUrl);
    const items = res.data.match(/<item>(.*?)<\/item>/gs);

    if (!items || items.length === 0) {
      console.log("❌ Tidak ditemukan postingan dalam feed.");
      return;
    }

    const urls = items.map(item => {
      const locMatch = item.match(/<link>(.*?)<\/link>/);
      const dateMatch = item.match(/<pubDate>(.*?)<\/pubDate>/);

      if (!locMatch || !dateMatch) return null;

      const loc = xmlEscape(locMatch[1]);
      const date = new Date(dateMatch[1]).toISOString().split('T')[0];

      return { loc, date };
    }).filter(Boolean);

    const xml = `<?xml version="1.0" encoding="UTF-8"?>
<urlset xmlns="http://www.sitemaps.org/schemas/sitemap/0.9">
${urls.map(u => `
  <url>
    <loc>${u.loc}</loc>
    <lastmod>${u.date}</lastmod>
    <changefreq>monthly</changefreq>
    <priority>0.7</priority>
  </url>`).join('')}
</urlset>`;

    fs.writeFileSync('sitemap.xml', xml);
    console.log('✅ sitemap.xml berhasil dibuat!');
  } catch (err) {
    console.error('❌ Gagal mengambil atau memproses feed:', err.message);
  }
})();
